  2. /**
  3.  * QPTPL is a template library for QueryPath.
  4.  *
  5.  * The QPTPL extension provides template tools that can be used in
  6.  * conjunction with QueryPath.
  7.  *
  8.  * There are two basic modes in which this tool operates. Both merge data into
  9.  * a pure HTML template. Both base their insertions on classes and IDs in the
  10.  * HTML data. Where they differ is in the kind of data merged into the template.
  11.  *
  12.  * One mode takes array data and does a deep (recursive) merge into the template.
  13.  * It can be used for simple substitutions, but it can also be used to loop through
  14.  * "rows" of data and create tables.
  15.  *
  16.  * The second mode takes a classed object and introspects that object to find out
  17.  * what CSS classes it is capable of filling. This is one way of bridging an object
  18.  * model and QueryPath data.
  19.  *
  20.  * The unit tests are a good place for documentation, as is the QueryPath webste.
  21.  *
  22.  * @package QueryPath
  23.  * @subpackage Extension
  24.  * @author M Butcher <matt@aleph-null.tv>
  25.  * @license http://opensource.org/licenses/lgpl-2.1.php LGPL or MIT-like license.
  26.  * @see QueryPathExtension
  27.  * @see QueryPathExtensionRegistry::extend()
  28.  * @see https://fedorahosted.org/querypath/wiki/QueryPathTemplate
  29.  */

  37.   /**
  38.    * Apply a template to an object and then insert the results.
  39.    *
  40.    * This takes a template (an arbitrary fragment of XML/HTML) and an object
  41.    * or array and inserts the contents of the object into the template. The
  42.    * template is then appended to all of the nodes in the current list.
  43.    *
  44.    * Note that the data in the object is *not* escaped before it is merged
  45.    * into the template. For that reason, an object can return markup (as
  46.    * long as it is well-formed).
  47.    * 
  48.    * @param mixed $template 
  49.    *   The template. It can be of any of the types that {@link qp()} supports
  50.    *   natively. Typically it is a string of XML/HTML.
  51.    * @param mixed $object 
  52.    *   Either an object or an associative array.
  53.    *   - In the case where the parameter
  54.    *   is an object, this will introspect the object, looking for getters (a la
  55.    *   Java bean behavior). It will then search the document for CSS classes
  56.    *   that match the method name. The function is then executed and its contents
  57.    *   inserted into the document. (If the function returns NULL, nothing is
  58.    *   inserted.)
  59.    *   - In the case where the paramter is an associative array, the function will
  60.    *   look through the template for CSS classes that match the keys of the
  61.    *   array. When an array key is found, the array value is inserted into the
  62.    *   DOM as a child of the currently matched element(s).
  63.    * @param array $options 
  64.    *   The options for this function. Valid options are:
  65.    *   - <None defined yet>
  66.    * @return QueryPath 
  67.    *   Returns a QueryPath object with all of the changes from the template
  68.    *   applied into the QueryPath elements.
  69.    * @see QueryPath::append()
  70.    */
  71.   public function tpl($template, $object, $options = array()) {
  72.     // Handle default options here.
  73.  
  74.     //$tqp = ($template instanceof QueryPath) ? clone $template: qp($template);
  75.     $tqp = qp($template);
  76.     
  77.     if (is_array($object) || $object instanceof Traversable) 
  78.       $this->tplArrayR($tqp, $object, $options);
  79.     elseif (is_object($object)) 
  80.       $this->tplObject($tqp, $object, $options);
  81.     
  82.     return $this->qp->append($tqp->top());
  83.   }

  90.    * @see tpl()

  93.    *   natively. Typically it is a string of XML/HTML.
  94.    * @param array $objects 
  95.    *   An indexed array containing a list of objects or arrays (See {@link tpl()})
  96.    *   that will be merged into the template.
  97.    * @param array $options

  99.    * @return QueryPath

  102.   public function tplAll($template, $objects, $options = array()) {

  164.   /**
  165.    * Introspect objects to map their functions to CSS classes in a template.
  166.    */
  167.   protected function tplObject($tqp, $object, $options = array()) {
  168.     $ref = new ReflectionObject($object);
  169.     $methods = $ref->getMethods();
  170.     foreach ($methods as $method) {
  171.       if (strpos($method->getName(), 'get') === 0) {
  172.         $cssClass = $this->method2class($method->getName());
  173.         if ($tqp->top()->find($cssClass)->size() > 0) {
  174.           $tqp->append($method->invoke($object));
  175.         }
  176.         else {
  177.           // Revert to the find() that found something.
  178.           $tqp->end();
  179.         }
  180.       }
  181.     }
  182.     //return $tqp->top();
  183.     return $tqp;
  184.   }

  186.   /**
  187.    * Recursively merge array data into a template.
  188.    */
  189.   public function tplArrayR($qp, $array, $options = NULL) {
  190.     if (!is_array($array) && !($array instanceof Traversable)) {
  191.       $qp->append($array);
  192.     }
  193.     elseif ($this->isAssoc($array)) {
  194.       // Do key/value substitutions
  195.       foreach ($array as $k => $v) {
  196.         
  197.         // If no dot or hash, assume class.
  198.         $first = substr($k,0,1);
  199.         if ($first != '.' && $first != '#') $k = '.' . $k;
  200.         
  201.         // If value is an array, recurse.
  202.         if (is_array($v))
  203.           $this->tplArrayR($qp->find($k), $v, $options);
  204.         else 
  205.           // Otherwise, try to append value.
  206.           $qp->branch()->children($k)->append($v);
  207.       }
  208.     }
  209.     else {
  210.       // Get a copy of the current template and then recurse.
  211.       foreach ($array as $entry) {
  212.         $eles = $qp->get();
  213.         $template = array();
  214.         
  215.         // We manually deep clone the template.
  216.         foreach ($eles as $ele) {
  217.           $template = $ele->cloneNode(TRUE);
  218.         }
  219.         $tpl = qp($template);
  220.         $tpl = $this->tplArrayR($tpl, $entry, $options);
  221.         $qp->before($tpl);
  222.       }
  223.       $qp->remove(); // Remove the original template.
  224.     }
  225.     return $qp;
  226.   }

  228.   /**
  229.    * Check whether an array is associative.
  230.    * If the keys of the array are not consecutive integers starting with 0,
  231.    * this will return false.
  232.    *
  233.    * @param array $array 
  234.    *   The array to test.
  235.    * @return Boolean 
  236.    *   TRUE if this is an associative array, FALSE otherwise.
  237.    */
  238.   public function isAssoc($array) {
  239.     $i = 0;
  240.     foreach ($array as $k => $v) if ($k !== $i++) return TRUE;
  241.     // If we get here, all keys passed.
  242.     return FALSE;
  243.   }

  251.    */

  253.     return '.' . substr($mname, 3);
