  2. /**
  3.  * Using QueryPath to Generate an RSS feed.
  4.  *
  5.  * This file contains an example of how QueryPath can be used
  6.  * to generate an RSS feed.
  7.  * @package Examples
  8.  * @author M Butcher <matt@aleph-null.tv>
  9.  * @license LGPL The GNU Lesser GPL (LGPL) or an MIT-like license.
  10.  */
  11.  
  12. require_once '../src/QueryPath/QueryPath.php';
  13.  
  14. // This is the stub RSS document.
  15. $rss_stub ='<?xml version="1.0"?>
  16. <rss version="2.0" 
  17.   xmlns:dc="http://purl.org/dc/elements/1.1/">
  18.   <channel>
  19.    <title></title>
  20.    <link></link>
  21.    <description></description>
  22.    <language>en</language>
  23.    <generator>QueryPath</generator>
  24.    </channel>
  25. </rss>
  26. ';
  27.  
  28. // This is the stub RSS element.
  29. $rss_item_stub = '<?xml version="1.0"?>
  30. <item>
  31.   <title>Untitled</title>
  32.   <link></link>
  33.   <description>
  34.   </description>
  35.   <comments></comments>
  36.   <category></category>
  37.   <pubDate></pubDate>
  38.   <guid isPermaLink="false"></guid>
  39. </item>';
  40.  
  41. // Here are some dummy items. For the same of 
  42. // simplicity, we are just using a nested array. Of
  43. // course, this could be a database lookup or whatever.
  44. $items = array(
  45.   array(
  46.     'title' => 'Item 1',
  47.     'link' => 'http://example.com/item1',
  48.     'description' => '<strong>This has embedded <em>HTML</em></strong>',
  49.     'comments' => 'http://example.com/item1/comments',
  50.     'category' => 'Some Term',
  51.     'pubDate' => date('r'),
  52.     'guid' => '123456-789',
  53.   ),
  54.   array(
  55.     'title' => 'Item 2',
  56.     'link' => 'http://example.com/item2',
  57.     'description' => '<strong>This has embedded <em>HTML</em></strong>',
  58.     'comments' => 'http://example.com/item2/comments',
  59.     'category' => 'Some Other Term',
  60.     'pubDate' => date('r'),
  61.     'guid' => '123456-790',
  62.   ),
  63. );
  64.  
  65. // The main QueryPath, which holds the channel.
  66. $qp = qp($rss_stub, 'title')
  67.   ->text('A QueryPath RSS Feed')
  68.   ->next('link')->text('http://example.com')
  69.   ->next('description')->text('QueryPath: Find your way.')
  70.   ->parent();
  71.  
  72. // For each element in the array above, we create a new 
  73. // QueryPath and then populate the XML fragment with data.
  74. foreach ($items as $item) {
  75.   
  76.   // Begin with the stub RSS item, with title currently selected.
  77.   $qpi = qp($rss_item_stub, 'title')
  78.     // Add a title.
  79.     ->text($item['title']) 
  80.     // Add a link. Note that we are giving no args to next() for the
  81.     // sake of simplicity.
  82.     ->next()->text($item['link'])
  83.     // Go to next element and add a description. Note that the text()
  84.     // call will automatically encode HTML. < will become &lt; and so on.
  85.     ->next()->text($item['description'])
  86.     // Go on down the list...
  87.     ->next()->text($item['comments'])
  88.     ->next()->text($item['category'])
  89.     ->next()->text($item['pubDate'])
  90.     ->next()->text($item['guid']);
  91.     
  92.   // Now we append it.
  93.   $qp->append($qpi->top());
  94. }
  95.  
  96. // If we were running this on a server, we would need to set the content
  97. // type:
  98. // header('Content-Type: application/rss+xml');
  99.  
  100. // Write the outpt as XML.
  101. $qp->writeXML();
  102. ?>
