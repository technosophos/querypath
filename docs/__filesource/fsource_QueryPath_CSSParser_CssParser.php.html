  21. /**
  22.  * An event handler for handling CSS 3 Selector parsing.
  23.  *
  24.  * This provides a standard interface for CSS 3 Selector event handling. As the
  25.  * parser parses a selector, it will fire events. Implementations of CssEventHandler
  26.  * can then handle the events.
  27.  *
  28.  * This library is inspired by the SAX2 API for parsing XML. Each component of a
  29.  * selector fires an event, passing the necessary data on to the event handler.
  30.  */
  31. interface CssEventHandler {
  32.   /** The is-exactly (=) operator. */
  33.   const isExactly = 0; // = 
  34.   /** The contains-with-space operator (~=). */
  35.   const containsWithSpace = 1; // ~=
  36.   /** The contains-with-hyphen operator (!=). */
  37.   const containsWithHyphen = 2; // |=
  38.   /** The contains-in-string operator (*=). */
  39.   const containsInString = 3; // *=
  40.   /** The begins-with operator (^=). */
  41.   const beginsWith = 4; // ^=
  42.   /** The ends-with operator ($=). */
  43.   const endsWith = 5; // $=
  44.   /** The any-element operator (*). */
  45.   const anyElement = '*';
  46.   
  47.   /**
  48.    * This event is fired when a CSS ID is encountered.
  49.    * An ID begins with an octothorp: #name.
  50.    *
  51.    * @param string $id 
  52.    *   The ID passed in.
  53.    */
  54.   public function elementID($id); // #name
  55.   /**
  56.    * Handle an element name.
  57.    * Example: name
  58.    * @param string $name 
  59.    *   The name of the element.
  60.    */
  61.   public function element($name); // name
  62.   /**
  63.    * Handle a namespaced element name.
  64.    * example: namespace|name
  65.    * @param string $name 
  66.    *   The tag name.
  67.    * @param string $namespace 
  68.    *   The namespace identifier (Not the URI)
  69.    */
  70.   public function elementNS($name, $namespace = NULL);
  71.   /**
  72.    * Handle an any-element (*) operator.
  73.    * Example: *
  74.    */
  75.   public function anyElement(); // *
  76.   /**
  77.    * Handle an any-element operator that is constrained to a namespace.
  78.    * Example: ns|*
  79.    * @param string $ns 
  80.    *   The namespace identifier (not the URI).
  81.    */
  82.   public function anyElementInNS($ns); // ns|*
  83.   /**
  84.    * Handle a CSS class selector.
  85.    * Example: .name
  86.    * @param string $name 
  87.    *   The name of the class.
  88.    */
  89.   public function elementClass($name); // .name
  90.   /**
  91.    * Handle an attribute selector.
  92.    * Example: [name=attr]
  93.    * Example: [name~=attr]
  94.    * @param string $name 
  95.    *   The attribute name.
  96.    * @param string $value 
  97.    *   The value of the attribute, if given.
  98.    * @param int $operation 
  99.    *   The operation to be used for matching. See {@link CssEventHandler}
  100.    *   constants for a list of supported operations.
  101.    */
  102.   public function attribute($name, $value = NULL, $operation = CssEventHandler::isExactly); // [name=attr]
  103.   /**
  104.    * Handle an attribute selector bound to a specific namespace.
  105.    * Example: [ns|name=attr]
  106.    * Example: [ns|name~=attr]
  107.    * @param string $name 
  108.    *   The attribute name.
  109.    * @param string $value 
  110.    *   The value of the attribute, if given.
  111.    * @param int $operation 
  112.    *   The operation to be used for matching. See {@link CssEventHandler}
  113.    *   constants for a list of supported operations.
  114.    * @param string $namespace 
  115.    *   The namespace identifier (not the URI).
  116.    */
  117.   public function attributeNS($name, $ns, $value = NULL, $operation = CssEventHandler::isExactly);
  118.   /**
  119.    * Handle a pseudo-class.
  120.    * Example: :name(value)
  121.    * @param string $name 
  122.    *   The pseudo-class name.
  123.    * @param string $value 
  124.    *   The value, if one is found.
  125.    */
  126.   public function pseudoClass($name, $value = NULL); //:name(value)
  127.   /**
  128.    * Handle a pseudo-element.
  129.    * Example: ::name
  130.    * @param string $name 
  131.    *   The pseudo-element name.
  132.    */
  133.   public function pseudoElement($name); // ::name
  134.   /**
  135.    * Handle a direct descendant combinator.
  136.    * Example: >
  137.    */
  138.   public function directDescendant(); // >
  139.   /**
  140.    * Handle a adjacent combinator.
  141.    * Example: +
  142.    */
  143.   public function adjacent(); // +
  144.   /**
  145.    * Handle an another-selector combinator.
  146.    * Example: ,
  147.    */
  148.   public function anotherSelector(); // ,
  149.   /**
  150.    * Handle a sibling combinator.
  151.    * Example: ~
  152.    */
  153.   public function sibling(); // ~ combinator
  154.   /**
  155.    * Handle an any-descendant combinator.
  156.    * Example: ' '
  157.    */
  158.   public function anyDescendant(); // ' ' (space) operator.
  159.   
  160. }

  162. /**
  163.  * Tokens for CSS.

  167. final class CssToken {
  168.   const char = 0;
  169.   const star = 1;
  170.   const rangle = 2;
  171.   const dot = 3;
  172.   const octo = 4;
  173.   const rsquare = 5;
  174.   const lsquare = 6;
  175.   const colon = 7;
  176.   const rparen = 8;
  177.   const lparen = 9;
  178.   const plus = 10;
  179.   const tilde = 11;
  180.   const eq = 12;
  181.   const pipe = 13;
  182.   const comma = 14;
  183.   const white = 15;
  184.   const quote = 16;
  185.   const squote = 17;
  186.   const bslash = 18;
  187.   const carat = 19;
  188.   const dollar = 20;
  189.   const at = 21; // This is not in the spec. Apparently, old broken CSS uses it.
  190.   
  191.   // In legal range for string.
  192.   const stringLegal = 99;
  193.   
  194.   /**
  195.    * Get a name for a given constant. Used for error handling.
  196.    */  
  197.   static function name($const_int) {
  198.     $a = array('character', 'star', 'right angle bracket', 
  199.       'dot', 'octothorp', 'right square bracket', 'left square bracket',
  200.       'colon', 'right parenthesis', 'left parenthesis', 'plus', 'tilde',
  201.       'equals', 'vertical bar', 'comma', 'space', 'quote', 'single quote',
  202.       'backslash', 'carat', 'dollar', 'at');
  203.     if (isset($a[$const_int])) {
  204.       return $a[$const_int];
  205.     }
  206.     elseif ($const_int == 99) {
  207.       return 'a legal non-alphanumeric character';
  208.     }
  209.     elseif ($const_int == FALSE) {
  210.       return 'end of file';
  211.     }
  212.     return sprintf('illegal character (%s)', $const_int);
  213.   }
  214. }

  216. /**
  217.  * Parse a CSS selector.
  218.  *
  219.  * In CSS, a selector is used to identify which element or elements
  220.  * in a DOM are being selected for the application of a particular style.
  221.  * Effectively, selectors function as a query language for a structured
  222.  * document -- almost always HTML or XML.
  223.  *
  224.  * This class provides an event-based parser for CSS selectors. It can be
  225.  * used, for example, as a basis for writing a DOM query engine based on
  226.  * CSS.
  227.  */
  228. class CssParser {
  229.   var $scanner = NULL;
  230.   var $buffer = '';
  231.   var $handler = NULL;
  232.   var $strict = FALSE;
  233.   
  234.   var $DEBUG = FALSE;
  235.   
  236.   /**
  237.    * Construct a new CSS parser object. This will attempt to
  238.    * parse the string as a CSS selector. As it parses, it will
  239.    * send events to the CssEventHandler implementation.
  240.    */
  241.   public function __construct($string, CssEventHandler $handler) {
  242.     $is = new CssInputStream($string);
  243.     $this->scanner = new CssScanner($is);
  244.     $this->handler = $handler;
  245.   }
  246.   
  247.   /**
  248.    * Parse the selector.
  249.    *
  250.    * This begins an event-based parsing process that will
  251.    * fire events as the selector is handled. A CssEventHandler
  252.    * implementation will be responsible for handling the events.
  253.    */
  254.   public function parse() /* throws CSSParseException */ {
  255.     $this->scanner->nextToken();
  256.     while ($this->scanner->token !== FALSE) {
  257.       if ($this->DEBUG) {
  258.         print "PARSE " . $this->scanner->token. "\n";
  259.       }
  260.       $this->selector();
  261.     }
  262.   }
  263.   
  264.   /**
  265.    * A restricted parser that can only parse simple selectors.
  266.    * The pseudoClass handler for this parser will throw an
  267.    * exception if it encounters a pseudo-element or the
  268.    * negation pseudo-class.
  269.    */
  270.   public function parseSimpleSelector() {
  271.     while ($this->scanner->token !== FALSE) {
  272.       if ($this->DEBUG) print "SIMPLE SELECTOR\n";
  273.       $this->allElements();
  274.       $this->elementName();
  275.       $this->elementClass();
  276.       $this->elementID();
  277.       $this->pseudoClass(TRUE); // Operate in restricted mode.
  278.       $this->attribute();
  279.       
  280.       // TODO: Need to add failure conditions here.
  281.     }
  282.   }
  283.   
  284.   /**
  285.    * Handle an entire CSS selector.
  286.    */
  287.   private function selector() {
  288.     if ($this->DEBUG) print "SELECTOR\n";
  289.     $this->consumeWhitespace(); // Remove leading whitespace
  290.     $this->simpleSelectors();
  291.     $this->combinator();
  292.   }
  293.   
  294.   /**
  295.    * Consume whitespace and return a count of the number of whitespace consumed.
  296.    */
  297.   private function consumeWhitespace() {
  298.     if ($this->DEBUG) print "CONSUME WHITESPACE\n";
  299.     $white = 0;
  300.     while ($this->scanner->token == CssToken::white) {
  301.       $this->scanner->nextToken();
  302.       ++$white;
  303.     }
  304.     return $white;
  305.   }
  306.   
  307.   /**
  308.    * Handle one of the five combinators: '>', '+', ' ', '~', and ','.
  309.    * This will call the appropriate event handlers.
  310.    * @see CssEventHandler::directDescendant(),
  311.    * @see CssEventHandler::adjacent(),
  312.    * @see CssEventHandler::anyDescendant(),
  313.    * @see CssEventHandler::anotherSelector().
  314.    */
  315.   private function combinator() {
  316.     if ($this->DEBUG) print "COMBINATOR\n";
  317.     /*
  318.      * Problem: ' ' and ' > ' are both valid combinators.
  319.      * So we have to track whitespace consumption to see
  320.      * if we are hitting the ' ' combinator or if the 
  321.      * selector just has whitespace padding another combinator.
  322.      */
  323.     
  324.     // Flag to indicate that post-checks need doing
  325.     $inCombinator = FALSE; 
  326.     $white = $this->consumeWhitespace();
  327.     $t = $this->scanner->token;    
  328.         
  329.     if ($t == CssToken::rangle) {
  330.       $this->handler->directDescendant();
  331.       $this->scanner->nextToken();
  332.       $inCombinator = TRUE;
  333.       //$this->simpleSelectors();
  334.     }
  335.     elseif ($t == CssToken::plus) {
  336.       $this->handler->adjacent();
  337.       $this->scanner->nextToken();
  338.       $inCombinator = TRUE;
  339.       //$this->simpleSelectors();
  340.     }
  341.     elseif ($t == CssToken::comma) {
  342.       $this->handler->anotherSelector();
  343.       $this->scanner->nextToken();
  344.       $inCombinator = TRUE;
  345.       //$this->scanner->selectors();
  346.     }
  347.     elseif ($t == CssToken::tilde) {
  348.       $this->handler->sibling();
  349.       $this->scanner->nextToken();
  350.       $inCombinator = TRUE;
  351.     }
  352.  
  353.     // Check that we don't get two combinators in a row.
  354.     if ($inCombinator) {
  355.       $white = 0;
  356.       if ($this->DEBUG) print "COMBINATOR: " . CssToken::name($t) . "\n";
  357.       $this->consumeWhitespace();
  358.       if ($this->isCombinator($this->scanner->token)) {
  359.         throw new CssParseException("Illegal combinator: Cannot have two combinators in sequence.");
  360.       }
  361.     }
  362.     // Check to see if we have whitespace combinator:
  363.     elseif ($white > 0) {
  364.       if ($this->DEBUG) print "COMBINATOR: any descendant\n";
  365.       $inCombinator = TRUE;
  366.       $this->handler->anyDescendant();
  367.     }
  368.     else {
  369.       if ($this->DEBUG) print "COMBINATOR: no combinator found.\n";
  370.     }
  371.     
  372.   }
  373.   
  374.   /**
  375.    * Check if the token is a combinator.
  376.    */
  377.   private function isCombinator($tok) {
  378.     $combinators = array(CssToken::plus, CssToken::rangle, CssToken::comma, CssToken::tilde);
  379.     return in_array($tok, $combinators);
  380.   }

  382.   /**
  383.    * Handle a simple selector.
  384.    */
  385.   private function simpleSelectors() {
  386.     if ($this->DEBUG) print "SIMPLE SELECTOR\n";
  387.     $this->allElements();
  388.     $this->elementName();
  389.     $this->elementClass();
  390.     $this->elementID();
  391.     $this->pseudoClass();
  392.     $this->attribute();
  393.   }
  394.   
  395.   /**
  396.    * Handles CSS ID selectors.
  397.    * This will call CssEventHandler::elementID().
  398.    */
  399.   private function elementID() {
  400.     if ($this->DEBUG) print "ELEMENT ID\n";
  401.     if ($this->scanner->token == CssToken::octo) {
  402.       $this->scanner->nextToken();
  403.       if ($this->scanner->token !== CssToken::char) {
  404.         throw new CssParseException("Expected string after #");
  405.       }
  406.       $id = $this->scanner->getNameString();
  407.       $this->handler->elementID($id);
  408.     }
  409.   }
  410.   
  411.   /**
  412.    * Handles CSS class selectors.
  413.    * This will call the CssEventHandler::elementClass() method.
  414.    */
  415.   private function elementClass() {
  416.     if ($this->DEBUG) print "ELEMENT CLASS\n";
  417.     if ($this->scanner->token == CssToken::dot) {
  418.       $this->scanner->nextToken();
  419.       $this->consumeWhitespace(); // We're very fault tolerent. This should prob through error.
  420.       $cssClass = $this->scanner->getNameString();
  421.       $this->handler->elementClass($cssClass);
  422.     }
  423.   }
  424.   
  425.   /**
  426.    * Handle a pseudo-class and pseudo-element.
  427.    *
  428.    * CSS 3 selectors support separate pseudo-elements, using :: instead
  429.    * of : for separator. This is now supported, and calls the pseudoElement
  430.    * handler, CssEventHandler::pseudoElement().
  431.    *
  432.    * This will call CssEventHandler::pseudoClass() when a
  433.    * pseudo-class is parsed.
  434.    */
  435.   private function pseudoClass($restricted = FALSE) {
  436.     if ($this->DEBUG) print "PSEUDO-CLASS\n";
  437.     if ($this->scanner->token == CssToken::colon) {
  438.  
  439.       // Check for CSS 3 pseudo element:
  440.       $isPseudoElement = FALSE;
  441.       if ($this->scanner->nextToken() === CssToken::colon) {
  442.         $isPseudoElement = TRUE;
  443.         $this->scanner->nextToken();
  444.       }
  445.       
  446.       $name = $this->scanner->getNameString();
  447.       if ($restricted && $name == 'not') {
  448.         throw new CssParseException("The 'not' pseudo-class is illegal in this context.");
  449.       }
  450.       
  451.       $value = NULL;
  452.       if ($this->scanner->token == CssToken::lparen) {
  453.         if ($isPseudoElement) {
  454.           throw new CssParseException("Illegal left paren. Pseudo-Element cannot have arguments.");
  455.         }
  456.         $value = $this->pseudoClassValue();
  457.       }
  458.       
  459.       // FIXME: This should throw errors when pseudo element has values.
  460.       if ($isPseudoElement) {
  461.         if ($restricted) {
  462.           throw new CssParseException("Pseudo-Elements are illegal in this context.");
  463.         }
  464.         $this->handler->pseudoElement($name);
  465.         $this->consumeWhitespace();
  466.         
  467.         // Per the spec, pseudo-elements must be the last items in a selector, so we 
  468.         // check to make sure that we are either at the end of the stream or that a
  469.         // new selector is starting. Only one pseudo-element is allowed per selector.
  470.         if ($this->scanner->token !== FALSE && $this->scanner->token !== CssToken::comma) {
  471.           throw new CssParseException("A Pseudo-Element must be the last item in a selector.");
  472.         }
  473.       } 
  474.       else {
  475.         $this->handler->pseudoClass($name, $value);
  476.       }
  477.     }
  478.   }
  479.   
  480.   /**
  481.    * Get the value of a pseudo-classes.
  482.    * 
  483.    * @return string 
  484.    *   Returns the value found from a pseudo-class.
  485.    *
  486.    * @todo Pseudoclasses can be passed pseudo-elements and
  487.    *   other pseudo-classes as values, which means :pseudo(::pseudo)
  488.    *   is legal.
  489.    */
  490.   private function pseudoClassValue() {
  491.     if ($this->scanner->token == CssToken::lparen) {
  492.       $buf = '';
  493.  
  494.       // For now, just leave pseudoClass value vague.
  495.       /*
  496.       // We have to peek to see if next char is a colon because
  497.       // pseudo-classes and pseudo-elements are legal strings here.
  498.       print $this->scanner->peek();
  499.       if ($this->scanner->peek() == ':') {
  500.         print "Is pseudo\n";
  501.         $this->scanner->nextToken();
  502.       
  503.         // Pseudo class
  504.         if ($this->scanner->token == CssToken::colon) {
  505.           $buf .= ':';
  506.           $this->scanner->nextToken();
  507.           // Pseudo element
  508.           if ($this->scanner->token == CssToken::colon) {
  509.             $buf .= ':';
  510.             $this->scanner->nextToken();
  511.           }
  512.           // Ident 
  513.           $buf .= $this->scanner->getNameString();
  514.         }
  515.       }
  516.       else {
  517.         print "fetching string.\n";
  518.         $buf .= $this->scanner->getQuotedString();
  519.         if ($this->scanner->token != CssToken::rparen) {
  520.           $this->throwError(CssToken::rparen, $this->scanner->token);
  521.         }
  522.         $this->scanner->nextToken();
  523.       }
  524.       return $buf;
  525.       */
  526.       $buf .= $this->scanner->getQuotedString();
  527.       return $buf;
  528.     }
  529.   }
  530.   
  531.   /**
  532.    * Handle element names.
  533.    * This will call the CssEventHandler::elementName().
  534.    *
  535.    * This handles:
  536.    * <code>
  537.    *  name (CssEventHandler::element())
  538.    *  |name (CssEventHandler::element())
  539.    *  ns|name (CssEventHandler::elementNS())
  540.    *  ns|* (CssEventHandler::elementNS())
  541.    * </code>
  542.    */
  543.   private function elementName() {
  544.     if ($this->DEBUG) print "ELEMENT NAME\n";
  545.     if ($this->scanner->token === CssToken::pipe) {
  546.       // We have '|name', which is equiv to 'name'
  547.       $this->scanner->nextToken();
  548.       $this->consumeWhitespace();
  549.       $elementName =  $this->scanner->getNameString();
  550.       $this->handler->element($elementName);
  551.     }
  552.     elseif ($this->scanner->token === CssToken::char) {
  553.       $elementName =  $this->scanner->getNameString();
  554.       if ($this->scanner->token == CssToken::pipe) {
  555.         // Get ns|name
  556.         $elementNS = $elementName;
  557.         $this->scanner->nextToken();
  558.         $this->consumeWhitespace();
  559.         if ($this->scanner->token === CssToken::star) {
  560.           // We have ns|* 
  561.           $this->handler->anyElementInNS($elementNS);
  562.           $this->scanner->nextToken();
  563.         }
  564.         elseif ($this->scanner->token !== CssToken::char) {
  565.           $this->throwError(CssToken::char, $this->scanner->token);
  566.         }
  567.         $elementName = $this->scanner->getNameString();
  568.         // We have ns|name
  569.         $this->handler->elementNS($elementName, $elementNS);
  570.       }
  571.       else {
  572.         $this->handler->element($elementName);
  573.       }
  574.     }
  575.   }
  576.   
  577.   /**
  578.    * Check for all elements designators. Due to the new CSS 3 namespace
  579.    * support, this is slightly more complicated, now, as it handles
  580.    * the *|name and *|* cases as well as *.
  581.    *
  582.    * Calls CssEventHandler::anyElement() or CssEventHandler::elementName().
  583.    */
  584.   private function allElements() {
  585.     if ($this->scanner->token === CssToken::star) {
  586.       $this->scanner->nextToken();
  587.       if ($this->scanner->token === CssToken::pipe) {
  588.         $this->scanner->nextToken();
  589.         if ($this->scanner->token === CssToken::star) {
  590.           // We got *|*. According to spec, this requires
  591.           // that the element has a namespace, so we pass it on
  592.           // to the handler:
  593.           $this->scanner->nextToken();
  594.           $this->handler->anyElementInNS('*');
  595.         }
  596.         else {
  597.           // We got *|name, which means the name MUST be in a namespce,
  598.           // so we pass this off to elementNameNS().
  599.           $name = $this->scanner->getNameString();
  600.           $this->handler->elementNS($name, '*');
  601.         }
  602.       }
  603.       else {
  604.         $this->handler->anyElement();
  605.       }
  606.     }
  607.   }
  608.   
  609.   /**
  610.    * Handler an attribute.
  611.    * An attribute can be in one of two forms:
  612.    * <code>[attrName]</code>
  613.    * or
  614.    * <code>[attrName="AttrValue"]</code>
  615.    *
  616.    * This may call the following event handlers: CssEventHandler::attribute().
  617.    */
  618.   private function attribute() {
  619.     if($this->scanner->token == CssToken::lsquare) {
  620.       $attrVal = $op = $ns = NULL;
  621.       
  622.       $this->scanner->nextToken();
  623.       $this->consumeWhitespace();
  624.       
  625.       if ($this->scanner->token === CssToken::at) {
  626.         if ($this->strict) {
  627.           throw new CssParseException('The @ is illegal in attributes.');
  628.         }
  629.         else {
  630.           $this->scanner->nextToken();
  631.           $this->consumeWhitespace();
  632.         }
  633.       }
  634.       
  635.       if ($this->scanner->token === CssToken::star) {
  636.         // Global namespace... requires that attr be prefixed,
  637.         // so we pass this on to a namespace handler.
  638.         $ns = '*';
  639.         $this->scanner->nextToken();
  640.       }
  641.       if ($this->scanner->token === CssToken::pipe) {
  642.         // Skip this. It's a global namespace.
  643.         $this->scanner->nextToken();
  644.         $this->consumeWhitespace();
  645.       }
  646.       
  647.       $attrName = $this->scanner->getNameString();
  648.       $this->consumeWhitespace();
  649.       
  650.       // Check for namespace attribute: ns|attr. We have to peek() to make 
  651.       // sure that we haven't hit the |= operator, which looks the same.
  652.       if ($this->scanner->token === CssToken::pipe && $this->scanner->peek() !== '=') {
  653.         // We have a namespaced attribute. 
  654.         $ns = $attrName;
  655.         $this->scanner->nextToken();
  656.         $attrName = $this->scanner->getNameString();
  657.         $this->consumeWhitespace();
  658.       }
  659.       
  660.       // Note: We require that operators do not have spaces 
  661.       // between characters, e.g. ~= , not ~ =.
  662.  
  663.       // Get the operator:
  664.       switch ($this->scanner->token) {
  665.         case CssToken::eq:
  666.           $this->consumeWhitespace();
  667.           $op = CssEventHandler::isExactly;
  668.           break;
  669.         case CssToken::tilde:
  670.           if ($this->scanner->nextToken() !== CssToken::eq) {
  671.             $this->throwError(CssToken::eq, $this->scanner->token);
  672.           }
  673.           $op = CssEventHandler::containsWithSpace;
  674.           break;
  675.         case CssToken::pipe:
  676.           if ($this->scanner->nextToken() !== CssToken::eq) {
  677.             $this->throwError(CssToken::eq, $this->scanner->token);
  678.           }
  679.           $op = CssEventHandler::containsWithHyphen;
  680.           break;
  681.         case CssToken::star:
  682.           if ($this->scanner->nextToken() !== CssToken::eq) {
  683.             $this->throwError(CssToken::eq, $this->scanner->token);
  684.           }
  685.           $op = CssEventHandler::containsInString;
  686.           break;
  687.         case CssToken::dollar;
  688.           if ($this->scanner->nextToken() !== CssToken::eq) {
  689.             $this->throwError(CssToken::eq, $this->scanner->token);
  690.           }
  691.           $op = CssEventHandler::endsWith;
  692.           break;
  693.         case CssToken::carat:
  694.           if ($this->scanner->nextToken() !== CssToken::eq) {
  695.             $this->throwError(CssToken::eq, $this->scanner->token);
  696.           }
  697.           $op = CssEventHandler::beginsWith;
  698.           break;
  699.       }
  700.       
  701.       if (isset($op)) {
  702.         // Consume '=' and go on.
  703.         $this->scanner->nextToken();
  704.         $this->consumeWhitespace();
  705.         
  706.         // So... here we have a problem. The grammer suggests that the
  707.         // value here is String1 or String2, both of which are enclosed
  708.         // in quotes of some sort, and both of which allow lots of special 
  709.         // characters. But the spec itself includes examples like this:
  710.         //   [lang=fr]
  711.         // So some bareword support is assumed. To get around this, we assume
  712.         // that bare words follow the NAME rules, while quoted strings follow
  713.         // the String1/String2 rules.
  714.         
  715.         if ($this->scanner->token === CssToken::quote || $this->scanner->token === CssToken::squote) {
  716.           $attrVal = $this->scanner->getQuotedString();
  717.         }
  718.         else {
  719.           $attrVal = $this->scanner->getNameString();
  720.         }
  721.         
  722.         if ($this->DEBUG) {
  723.           print "ATTR: $attrVal AND OP: $op\n";
  724.         }
  725.       }
  726.       
  727.       $this->consumeWhitespace();
  728.       
  729.       if ($this->scanner->token != CssToken::rsquare) {
  730.         $this->throwError(CssToken::rsquare, $this->scanner->token);
  731.       }
  732.       
  733.       if (isset($ns)) {
  734.         $this->handler->attributeNS($attrName, $ns, $attrVal, $op);
  735.       }
  736.       elseif ($attrVal) {
  737.         $this->handler->attribute($attrName, $attrVal, $op);
  738.       }
  739.       else {
  740.         $this->handler->attribute($attrName);
  741.       }
  742.       $this->scanner->nextToken();
  743.     }
  744.   }
  745.   
  746.   /**
  747.    * Utility for throwing a consistantly-formatted parse error.
  748.    */
  749.   private function throwError($expected, $got) {
  750.     $filter = sprintf('Expected %s, got %s', CssToken::name($expected), CssToken::name($got));
  751.     throw new CssParseException($filter);
  752.   }
  753.   
  754. }
  755.  
  756. /**
  757.  * Scanner for CSS selector parsing.
  758.  *
  759.  * This provides a simple scanner for traversing an input stream.
  760.  */
  761. final class CssScanner {
  762.   var $is = NULL;
  763.   public $value = NULL;
  764.   public $token = NULL;
  765.   
  766.   var $recurse = FALSE;
  767.   var $it = 0;
  768.   
  769.   /**
  770.    * Given a new input stream, tokenize the CSS selector string.
  771.    * @see CssInputStream
  772.    * @param CssInputStream $in 
  773.    *   An input stream to be scanned.
  774.    */
  775.   public function __construct(CssInputStream $in) {
  776.     $this->is = $in;
  777.   }
  778.   
  779.   /**
  780.    * See the next char without removing it from the stack.
  781.    *
  782.    * @return char 
  783.    *   Returns the next character on the stack.
  784.    */
  785.   public function peek() {
  786.     return $this->is->peek();
  787.   }
  788.   
  789.   /**
  790.    * Get the next token in the input stream.
  791.    *
  792.    * This sets the current token to the value of the next token in
  793.    * the stream.
  794.    *
  795.    * @return int 
  796.    *   Returns an int value corresponding to one of the CssToken constants,
  797.    *   or FALSE if the end of the string is reached. (Remember to use
  798.    *   strong equality checking on FALSE, since 0 is a valid token id.)
  799.    */
  800.   public function nextToken() {
  801.     $tok = -1;
  802.     ++$this->it;
  803.     if ($this->is->isEmpty()) {
  804.       if ($this->recurse) {
  805.         throw new Exception("Recursion error detected at iteration " . $this->it . '.');
  806.         exit();
  807.       }
  808.       //print "{$this->it}: All done\n";
  809.       $this->recurse = TRUE;
  810.       $this->token = FALSE;
  811.       return FALSE;
  812.     }
  813.     $ch = $this->is->consume();
  814.     //print __FUNCTION__ . " Testing $ch.\n";
  815.     if (ctype_space($ch)) {
  816.       $this->value = ' '; // Collapse all WS to a space.
  817.       $this->token = $tok = CssToken::white;
  818.       //$ch = $this->is->consume();
  819.       return $tok;
  820.     }
  821.     
  822.     if (ctype_alnum($ch) || $ch == '-' || $ch == '_') {
  823.       // It's a character
  824.       $this->value = $ch; //strtolower($ch);
  825.       $this->token = $tok = CssToken::char;
  826.       return $tok;
  827.     }
  828.     
  829.     $this->value = $ch;
  830.     
  831.     switch($ch) {
  832.       case '*':
  833.         $tok = CssToken::star;
  834.         break;
  835.       case chr(ord('>')):
  836.         $tok = CssToken::rangle;
  837.         break;
  838.       case '.':
  839.         $tok = CssToken::dot;
  840.         break;
  841.       case '#':
  842.         $tok = CssToken::octo;
  843.         break;
  844.       case '[':
  845.         $tok = CssToken::lsquare;
  846.         break;
  847.       case ']':
  848.         $tok = CssToken::rsquare;
  849.         break;
  850.       case ':':
  851.         $tok = CssToken::colon;
  852.         break;
  853.       case '(':
  854.         $tok = CssToken::lparen;
  855.         break;
  856.       case ')':
  857.         $tok = CssToken::rparen;
  858.         break;
  859.       case '+':
  860.         $tok = CssToken::plus;
  861.         break;
  862.       case '~':
  863.         $tok = CssToken::tilde;
  864.         break;
  865.       case '=':
  866.         $tok = CssToken::eq;
  867.         break;
  868.       case '|':
  869.         $tok = CssToken::pipe;
  870.         break;
  871.       case ',':
  872.         $tok = CssToken::comma;
  873.         break;
  874.       case chr(34):
  875.         $tok = CssToken::quote;
  876.         break;
  877.       case "'":
  878.         $tok = CssToken::squote;
  879.         break;
  880.       case '\\':
  881.         $tok = CssToken::bslash;
  882.         break;
  883.       case '^':
  884.         $tok = CssToken::carat;
  885.         break;
  886.       case '$':
  887.         $tok = CssToken::dollar;
  888.         break;
  889.       case '@':
  890.         $tok = CssToken::at;
  891.         break;
  892.     }
  893.     
  894.     
  895.     // Catch all characters that are legal within strings.
  896.     if ($tok == -1) {
  897.       // TODO: This should be UTF-8 compatible, but PHP doesn't
  898.       // have a native UTF-8 string. Should we use external
  899.       // mbstring library?
  900.  
  901.       $ord = ord($ch);
  902.       // Characters in this pool are legal for use inside of
  903.       // certain strings. Extended ASCII is used here, though I
  904.       // Don't know if these are really legal.

  953.

  963.         }
  964.         elseif ($escape) {
  965.           // Turn off escaping
  966.           $buf .= $this->value;

  968.         }
  969.         elseif ($this->token === $end) {
  970.           // At end of string; skip token and break.
  971.           $this->nextToken();

  973.         }
  974.         else {
  975.           // Append char.
  976.           $buf .= $this->value;
  977.         }
  978.         $this->nextToken();
  979.       }
  980.       return $buf;
  981.     }
  982.   }
