  3.  * Provide QueryPath with XSLT support using the PHP libxslt module.
  4.  *
  5.  * This is called 'QPXSL' instead of 'QPXSLT' in accordance with the name
  6.  * of the PHP extension that provides libxslt support.
  7.  *
  8.  * You must have PHP XSL support for this to function.
  9.  *
  10.  * @package QueryPath
  11.  * @subpackage Extension
  12.  * @author M Butcher <matt@aleph-null.tv>
  13.  * @license http://opensource.org/licenses/lgpl-2.1.php LGPL or MIT-like license.
  14.  * @see QueryPathExtension
  15.  * @see QueryPathExtensionRegistry::extend()
  16.  * @see QPXSL
  17.  * @see QPXML
  18.  */
  19.  
  20. /**
  21.  * Provide tools for running XSL Transformation (XSLT) on a document.
  22.  *
  23.  * This extension provides the {@link QPXSL::xslt()} function, which transforms
  24.  * a source XML document into another XML document according to the rules in
  25.  * an XSLT document.
  26.  *
  27.  * This QueryPath extension can be used as follows:
  28.  * <code>
  29.  * <?php
  30.  * require 'QueryPath/QueryPath.php';
  31.  * require 'QueryPath/Extension/QPXSL.php';
  32.  *
  33.  * qp('src.xml')->xslt('stylesheet.xml')->writeXML();
  34.  * ?>
  35.  *
  36.  * This will transform src.xml according to the XSLT rules in
  37.  * stylesheet.xml. The results are returned as a QueryPath object, which
  38.  * is written to XML using {@link QueryPath::writeXML()}.
  39.  * </code>
  40.  */
  41. class QPXSL implements QueryPathExtension {
  42.   
  43.   protected $src = NULL;
  44.   
  45.   public function __construct(QueryPath $qp) {
  46.     $this->src = $qp;
  47.   }
  48.   
  49.   /**
  50.    * Given an XSLT stylesheet, run a transformation.
  51.    *
  52.    * This will attempt to read the provided stylesheet and then
  53.    * execute it on the current source document.
  54.    *
  55.    * @param mixed $style 
  56.    *   This takes a QueryPath object or <em>any</em> of the types that the
  57.    *   {@link qp()} function can take.
  58.    * @return QueryPath 
  59.    *   A QueryPath object wrapping the transformed document. Not that this is a
  60.    *   <i>different</em> document than the original. As such, it has no history.
  61.    *   You cannot call {@link QueryPath::end()} to undo a transformation. (However,
  62.    *   the original source document will remain unchanged.)
  63.    */
  64.   public function xslt($style) {
  65.     if (!($style instanceof QueryPath)) {
  66.       $style = qp($style);
  67.     }
  68.     $sourceDoc = $this->src->top()->get(0)->ownerDocument;
  69.     $styleDoc = $style->get(0)->ownerDocument;
  70.     $processor = new XSLTProcessor();
  71.     $processor->importStylesheet($styleDoc);
  72.     return qp($processor->transformToDoc($sourceDoc));
  73.   }
  74. }
