  424.             foreach ($data->childNodes as $n)
  425.               $m->appendChild($n->cloneNode(TRUE));
  426.           }
  427.           else {
  428.             // Otherwise a standard clone will do.
  429.             $m->appendChild($data->cloneNode(TRUE));
  430.           }
  431.           
  432.         }
  433.       }
  434.         
  435.     }
  436.     return $this;

  589.   /**
  590.    * A depth-checking function. Typically, it only needs to be
  591.    * invoked with the first parameter. The rest are used for recursion.
  592.    * @see deepest();
  593.    */
  594.   protected function deepestNode(DOMNode $ele, $depth = 0, $current = NULL, &$deepest = NULL) {
  595.     if (!isset($current)) $current = array($ele);
  596.     if (!isset($deepest)) $deepest = $depth;
  597.     if ($ele->hasChildNodes()) {
  598.       foreach ($ele->childNodes as $child) {
  599.         if ($child->nodeType === XML_ELEMENT_NODE) {
  600.           $current = $this->deepestNode($child, $depth + 1, $current, $deepest);
  601.         }
  602.       }
  603.     }
  604.     elseif ($depth > $deepest) {
  605.       $current = array($ele);
  606.       $deepest = $depth;
  607.     }
  608.     elseif ($depth === $deepest) {
  609.       $current[] = $ele;
  610.     }
  611.     return $current;
  612.   }

  614.   /**

  619.    * - If item is a string, this is converted into a document fragment and returned.
  620.    * - If item is a QueryPath, then the first item is retrieved and this call function
  621.    *   is called recursivel.
  622.    * - If the item is a DOMNode, it is imported into the current DOM if necessary.
  623.    * - If the item is a SimpleXMLElement, it is converted into a DOM node and then
  624.    *   imported.

  626.   protected function prepareInsert($item) {
  627.     if(empty($item)) {
  628.       return;
  629.     }
  630.     elseif (is_string($item)) {

  632.       if ($this->isXMLish($item)) {

  635.         return $frag;
  636.       }

  638.         return $this->document->createElement($item);
  639.       }
  640.       */
  641.       $frag = $this->document->createDocumentFragment();
  642.       $frag->appendXML($item);
  643.       return $frag;
  644.     }
  645.     elseif ($item instanceof QueryPath) {
  646.       if ($item->size() == 0) 
  647.         return;
  648.         
  649.       return $this->prepareInsert($item->get(0));

  662.     // What should we do here?
  663.     //var_dump($item);

  665.   }
