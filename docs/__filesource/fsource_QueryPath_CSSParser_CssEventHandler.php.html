  1. <?php
  2. /**
  3.  * This file contains a full implementation of the CssEventHandler interface.
  4.  * 
  5.  * The tools in this package initiate a CSS selector parsing routine and then
  6.  * handle all of the callbacks.
  7.  *
  8.  * The implementation provided herein adheres to the CSS 3 Selector specification
  9.  * with the following caveats:
  10.  *
  11.  *  - The negation (:not()) and containment (:has()) pseudo-classes allow *full*
  12.  *    selectors and not just simple selectors.
  13.  *  - There are a variety of additional pseudo-classes supported by this
  14.  *    implementation that are not part of the spec. Most of the jQuery
  15.  *    pseudo-classes are supported. The :x-root pseudo-class is also supported.
  16.  *  - Pseudo-classes that require a User Agent to function have been disabled.
  17.  *    Thus there is no :hover pseudo-class.
  18.  *  - All pseudo-elements require the double-colon (::) notation. This breaks
  19.  *    backward compatibility with the 2.1 spec, but it makes visible the issue
  20.  *    that pseudo-elements cannot be effectively used with most of the present
  21.  *    library. They return strings instead of elements.
  22.  *  - The pseudo-classes first-of-type, nth-of-type and last-of-type may or may
  23.  *    not conform to the specification. The spec is unclear.
  24.  *  - pseudo-class filters of the form -an+b do not function as described in the
  25.  *    specification. However, they do behave the same way here as they do in
  26.  *    jQuery.
  27.  *  - This library DOES provide XML namespace aware tools. Selectors can use
  28.  *    namespaces to increase specificity.
  29.  *  - This library does nothing with the CSS 3 Selector specificity rating. Of
  30.  *    course specificity is preserved (to the best of our abilities), but there
  31.  *    is no calculation done.
  32.  *
  33.  * For detailed examples of how the code works and what selectors are supported,
  34.  * see the {@see CssEventTests.php} file, which contains the unit tests used for
  35.  * testing this implementation.
  36.  *
  37.  * @package QueryPath
  38.  * @subpackage CSSParser
  39.  */

  46. /**
  47.  * Handler that tracks progress of a query through a DOM.
  48.  *
  49.  * The main idea is that we keep a copy of the tree, and then use an
  50.  * array to keep track of matches. To handle a list of selectors (using
  51.  * the comma separator), we have to track both the currently progressing
  52.  * match and the previously matched elements.
  53.  *
  54.  * To use this handler:
  55.  * <code>
  56.  * $filter = '#id'; // Some CSS selector
  57.  * $handler = new QueryPathCssParser(DOMNode $dom);
  58.  * $parser = new CssParser();
  59.  * $parser->parse($filter, $handler);
  60.  * $matches = $handler->getMatches();
  61.  * </code>
  62.  *
  63.  * $matches will be an array of zero or more DOMElement objects.
  64.  */

  75.   public function __construct($dom) {
  76.     // Array of DOMElements
  77.     if (is_array($dom)) {
  78.       $matches = array();
  79.       foreach($dom as $item) {
  80.         if ($item instanceof DOMNode && $item->nodeType == XML_ELEMENT_NODE) {
  81.           $matches[] = $item;
  82.         }
  83.       }
  84.       $this->dom = count($matches) > 0 ? $matches[0] : NULL;
  85.       $this->matches = $matches;
  86.     }
  87.     // DOM Document -- we get the root element.
  88.     elseif ($dom instanceof DOMDocument) {
  89.       $this->dom = $dom->documentElement;
  90.       $this->matches = array($dom->documentElement);
  91.     }
  92.     // DOM Element -- we use this directly
  93.     elseif ($dom instanceof DOMElement) {
  94.       $this->dom = $dom;
  95.       $this->matches = array($dom);
  96.     }
  97.     // NodeList -- We turn this into an array
  98.     elseif ($dom instanceof DOMNodeList) {
  99.       $matches = array();
  100.       foreach ($dom as $item) {
  101.         if ($item->nodeType == XML_ELEMENT_NODE) {
  102.           $matches[] = $item;
  103.         }
  104.       }
  105.       $this->dom = $matches;
  106.       $this->matches = $matches;
  107.     }
  108.     // FIXME: Handle SimpleXML!
  109.     // Uh-oh... we don't support anything else.
  110.     else {
  111.       throw new Exception("Unhandled type: " . get_class($dom));
  112.     }

  235.   public function attribute($name, $value = NULL, $operation = CssEventHandler::isExactly) {

  244.           }
  245.         }
  246.         else {
  247.           // If no value exists, then we consider it a match.
  248.           $found[] = $item;
  249.         }
  250.       }
  251.     }

  310.   /**

  316.     $name = strtolower($name);
  317.     // Need to handle known pseudoclasses.
  318.     switch($name) {
  319.       case 'visited':
  320.       case 'hover':
  321.       case 'active':
  322.       case 'focus':
  323.       case 'animated': //  Last 3 are from jQuery
  324.       case 'visible':
  325.       case 'hidden':
  326.         // These require a UA, which we don't have.
  327.       case 'target':
  328.         // This requires a location URL, which we don't have.
  329.         $this->matches = array();
  330.         break;
  331.       case 'indeterminate':
  332.         // The assumption is that there is a UA and the format is HTML.
  333.         // I don't know if this should is useful without a UA.
  334.         throw new NotImplementedException(":indeterminate is not implemented.");
  335.         break;
  336.       case 'lang':
  337.         // No value = exception.
  338.         if (!isset($value)) {
  339.           throw new NotImplementedException("No handler for lang pseudoclass without value.");
  340.         }
  341.         $this->lang($value);
  342.         break;
  343.       case 'link':
  344.         $this->searchForAttr('href');
  345.         break;
  346.       case 'root':
  347.         if (empty($this->dom)) {
  348.           $this->matches = array();
  349.         }
  350.         elseif (is_array($this->dom)) {
  351.           $this->matches = array($this->dom[0]->ownerDocument->documentElement);
  352.         }
  353.         elseif ($this->dom instanceof DOMNode) {
  354.           $this->matches = array($this->dom->ownerDocument->documentElement);
  355.         }
  356.         elseif ($this->dom instanceof DOMNodeList && $this->dom->length > 0) {
  357.           $this->matches = array($this->dom->item(0)->ownerDocument->documentElement);
  358.         }
  359.         else {
  360.           // Hopefully we never get here:
  361.           $this->matches = array($this->dom);
  362.         }
  363.         break;
  364.       
  365.       // NON-STANDARD extensions for reseting to the "top" items set in
  366.       // the constructor.  
  367.       case 'x-root':
  368.       case 'x-reset':
  369.         $this->matches = array($this->dom);
  370.         break;        
  371.       
  372.       // NON-STANDARD extensions for simple support of even and odd. These
  373.       // are supported by jQuery, FF, and other user agents.  
  374.       case 'even':
  375.         $this->nthChild(2, 0);
  376.         break;
  377.       case 'odd':

  394.       case 'nth-last-of-type':

  436.       case 'parent':
  437.         $matches = $this->candidateList();

  1112.     
  1113.     // According to the spec:
